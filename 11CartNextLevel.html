function parsePrice(str) {
 if (!str) return 0;
 return parseFloat(str.toString().replace(/,/g,"").replace(/[^\d.]/g,"").trim()) || 0;
}
function formatPrice(num) {
 if (!num) return "0.00";
 return num.toLocaleString("en-LK", {minimumFractionDigits:2, maximumFractionDigits:2});
}
function col(item, name) {
 if (item[name] !== undefined) return item[name] || "";
 const low = name.toLowerCase().replace(/\s+/g," ").trim();
 for (const k of Object.keys(item)) {
   const kc = k.replace(/^\uFEFF/,"").toLowerCase().replace(/\s+/g," ").trim();
   if (kc === low) return item[k] || "";
 }
 return "";
}

           discountedPrice: parsePrice(col(item,"Discounted Price")),
           description: col(item,"Description").trim(),
           brand: col(item,"Brand").trim(),
           stock: parseInt(col(item,"Stock")) || 0,
           thumbnail: col(item,"Thumbnail").trim(),
           image1: col(item,"Image1").trim(),
           image2: col(item,"Image2").trim(),
           ageRange: col(item,"Age Range").trim(),
           material: col(item,"Material").trim(),
           theme: col(item,"Theme").trim(),
         });
       });
       if (!allProducts.length) {
         container.innerHTML = "<div style='grid-column:1/-1;text-align:center;padding:60px;font-size:18px;font-weight:700;color:#888'>No products available right now.</div>";
         return;
       }
       renderProducts(allProducts);
       updateCartUI(); // Initialize cart counts
     }
   });
 } catch(err) {
   container.innerHTML = `<div style='grid-column:1/-1;text-align:center;padding:60px'>
     <div style='font-size:48px; color:var(--orange)'><i class="fas fa-exclamation-circle"></i></div>
     <p style='font-size:20px;font-weight:800;margin-top:16px; color:var(--dark)'>Unable to load products</p>
     <p style='font-size:15px;color:#888;margin-top:8px'>${err.message}</p>
   </div>`;
 }
}

/* ===== RENDER PRODUCTS ===== */
function renderProducts(products) {
 renderedProducts = products;
 const container = document.getElementById('productsContainer');
 document.getElementById('productCount').innerHTML = `<b>${products.length}</b> items found`;

 if (!products.length) {
   container.innerHTML = `
     <div style='grid-column:1/-1;text-align:center;padding:80px 20px; background:white; border-radius:var(--radius-xl); border:1px dashed #ccc;'>
       <i class="fas fa-search" style="font-size:48px; color:#ddd; margin-bottom:16px;"></i>
       <h3 style="font-size:20px; color:var(--dark); margin-bottom:8px;">No products found</h3>
       <p style="color:#888;">Try adjusting your filters or search term.</p>
       <button class="btn-primary" style="margin-top:20px;" onclick="resetFilters()">Clear Filters</button>
     </div>`;
   return;
 }

 let html = "";
 for (let i = 0; i < products.length; i++) {
   const p = products[i];
   const hasDis = p.discountedPrice > 0 && p.discountedPrice < p.price;
   const showPrice = hasDis ? p.discountedPrice : p.price;
   const discPct = hasDis ? Math.round(((p.price - p.discountedPrice) / p.price) * 100) : 0;
   const lowStock = p.stock > 0 && p.stock <= 5;
   const thumb = p.thumbnail || p.image1 || "https://via.placeholder.com/400x400?text=11Cart";
   const safeName = p.name.replace(/"/g,""").replace(/'/g,"'");

   html += `<div class="product-card" onclick="openModal(${i})">`;
   html += `<div class="badge-row">`;
   html += lowStock ? `<span class="badge-stock"><i class="fas fa-bolt"></i> Only ${p.stock} left!</span>` : `<span></span>`;
   html += hasDis ? `<span class="badge-discount">-${discPct}% OFF</span>` : '';
   html += `</div>`;
   html += `<div class="card-img-wrap"><img loading="lazy" src="${thumb}" alt="${safeName}" onerror="this.src='https://via.placeholder.com/400x400?text=11Cart'"></div>`;
   html += `<div class="card-body">`;
   html += `<span class="card-cat">${p.category}</span>`;
   html += `<h3>${p.name}</h3>`;
   html += `<div class="card-price-row"><span class="card-price">Rs ${formatPrice(showPrice)}</span>`;
   if (hasDis) html += `<span class="card-old-price">Rs ${formatPrice(p.price)}</span>`;
   html += `</div>`;
   html += `<div class="card-actions">`;
   html += `<button class="btn-quick-view" onclick="event.stopPropagation();openModal(${i})" title="Quick View"><i class="fas fa-eye"></i></button>`;
   html += `<button class="btn-add-cart" onclick="event.stopPropagation();addToCartObj(${i})"><i class="fas fa-cart-plus"></i> Add to Cart</button>`;
   html += `</div></div></div>`;
 }
 container.innerHTML = html;
}

/* ===== FILTER & SORT ===== */
function applyAndRender() {
 let list = [...allProducts];
 if (activeCategory) list = list.filter(p => p.category.toLowerCase().includes(activeCategory.toLowerCase()));
 if (searchVal) list = list.filter(p => p.name.toLowerCase().includes(searchVal) || p.category.toLowerCase().includes(searchVal));
 const sort = document.getElementById('sortSelect').value;
 if (sort === "low") list.sort((a,b) => (a.discountedPrice||a.price)-(b.discountedPrice||b.price));
 else if (sort === "high") list.sort((a,b) => (b.discountedPrice||b.price)-(a.discountedPrice||a.price));
 else if (sort === "name") list.sort((a,b) => a.name.localeCompare(b.name));
 renderProducts(list);
}

function filterCategory(cat, el) {
 activeCategory = cat;
 document.querySelectorAll('.category-card').forEach(c => c.classList.remove('active'));
 if (el) el.classList.add('active');
 applyAndRender();
 document.getElementById('products').scrollIntoView({behavior:'smooth', block: 'start'});
}

function resetFilters() {
 activeCategory = null;
 searchVal = "";
 document.getElementById('searchInput').value = "";
 document.getElementById('sortSelect').value = "";
 document.querySelectorAll('.category-card').forEach(c => c.classList.remove('active'));
 renderProducts(allProducts);
}

document.getElementById('searchInput').addEventListener('input', function() {
 searchVal = this.value.toLowerCase();
 applyAndRender();
});
document.getElementById('sortSelect').addEventListener('change', applyAndRender);


/* ===== SHOPPING CART LOGIC ===== */

function saveCart() {
 localStorage.setItem('11cart_data', JSON.stringify(cart));
}

function getProductPrice(p) {
 return (p.discountedPrice > 0 && p.discountedPrice < p.price) ? p.discountedPrice : p.price;
}

// Add via global index (from rendered grid)
function addToCartObj(renderedIndex) {
 const p = renderedProducts[renderedIndex];
 if(!p) return;
 addItemToCart(p);
}

// Add directly via product object
function addItemToCart(p) {
 const existing = cart.find(item => item.id === p.id);
 if (existing) {
   existing.qty += 1;
 } else {
   cart.push({ ...p, qty: 1 });
 }
 
 if (typeof fbq !== 'undefined') {
   fbq('track', 'AddToCart', { content_name: p.name, value: getProductPrice(p), currency: 'LKR' });
 }

 saveCart();
 updateCartUI();
 showToast(`🛒 Added ${p.name} to cart`);
 
 // Bump animation on badge
 const badges = document.querySelectorAll('.cart-badge');
 badges.forEach(b => {
   b.classList.remove('bump');
   void b.offsetWidth; // trigger reflow
   b.classList.add('bump');
 });
}

function changeQty(id, delta) {
 const item = cart.find(i => i.id === id);
 if (!item) return;
 
 item.qty += delta;
 if (item.qty <= 0) {
   cart = cart.filter(i => i.id !== id);
 }
 saveCart();
 updateCartUI();
}

function removeFromCart(id) {
 cart = cart.filter(i => i.id !== id);
 saveCart();
 updateCartUI();
}

function updateCartUI() {
 // Update badges
 const totalItems = cart.reduce((sum, item) => sum + item.qty, 0);
 document.getElementById('navCartBadge').textContent = totalItems;
 document.getElementById('floatCartBadge').textContent = totalItems;

 const container = document.getElementById('cartItemsContainer');
 const totalEl = document.getElementById('cartTotalValue');
 
 if (cart.length === 0) {
   container.innerHTML = `
     <div class="empty-cart">
       <i class="fas fa-shopping-basket"></i>
       <h3>Your cart is empty</h3>
       <p>Looks like you haven't added any items yet.</p>
       <button class="btn-primary" style="margin-top:20px" onclick="toggleCart(); document.getElementById('products').scrollIntoView({behavior:'smooth'})">Continue Shopping</button>
     </div>`;
   totalEl.textContent = "Rs 0.00";
   return;
 }

 let html = "";
 let totalAmount = 0;

 cart.forEach(item => {
   const price = getProductPrice(item);
   totalAmount += price * item.qty;
   const thumb = item.thumbnail || item.image1 || "https://via.placeholder.com/100?text=Item";
   const safeName = item.name.replace(/"/g,""");

   html += `
     <div class="cart-item">
       <img src="${thumb}" alt="${safeName}" class="cart-item-img">
       <div class="cart-item-info">
         <div class="cart-item-title">${item.name}</div>
         <div class="cart-item-price">Rs ${formatPrice(price)}</div>
         <div class="cart-qty-ctrl">
           <button class="qty-btn" onclick="changeQty(${item.id}, -1)"><i class="fas fa-minus" style="font-size:10px"></i></button>
           <span class="qty-num">${item.qty}</span>
           <button class="qty-btn" onclick="changeQty(${item.id}, 1)"><i class="fas fa-plus" style="font-size:10px"></i></button>
         </div>
       </div>
       <button class="cart-item-remove" onclick="removeFromCart(${item.id})" title="Remove item"><i class="fas fa-trash-alt"></i></button>
     </div>
   `;
 });

 container.innerHTML = html;
 totalEl.textContent = "Rs " + formatPrice(totalAmount);
}

function toggleCart() {
 const overlay = document.getElementById('cartOverlay');
 const drawer = document.getElementById('cartDrawer');
 const isOpen = overlay.classList.contains('open');
 
 if (isOpen) {
   overlay.classList.remove('open');
   drawer.classList.remove('open');
   document.body.style.overflow = '';
 } else {
   overlay.classList.add('open');
   drawer.classList.add('open');
   document.body.style.overflow = 'hidden';
   updateCartUI(); // Ensure fresh render
 }
}

function closeCartBg(e) {
 if (e.target === document.getElementById('cartOverlay')) toggleCart();
}

function checkoutWhatsApp() {
 if (cart.length === 0) {
   showToast("⚠️ Your cart is empty!");
   return;
 }

 let text = "Hello 11Cart! 👋\nI would like to place an order:\n\n";
 let total = 0;

 cart.forEach((item, index) => {
   const price = getProductPrice(item);
   total += price * item.qty;
   text += `${index + 1}. *${item.name}*\n`;
   text += `   Qty: ${item.qty} x Rs ${formatPrice(price)}\n`;
 });

 text += `\n*Total Amount: Rs ${formatPrice(total)}*\n\n`;
 text += "Please let me know the delivery details. Thank you!";

 const encodedText = encodeURIComponent(text);
 const waUrl = `https://wa.me/94768665158?text=${encodedText}`;
 
 if (typeof fbq !== 'undefined') {
   fbq('track', 'InitiateCheckout');
 }

 window.open(waUrl, '_blank');
}

/* ===== TOAST NOTIFICATIONS ===== */
function showToast(message) {
 const container = document.getElementById('toastContainer');
 const toast = document.createElement('div');
 toast.className = 'toast';
 toast.innerHTML = `<span>${message}</span>`;
 container.appendChild(toast);
 
 setTimeout(() => {
   toast.classList.add('hide');
   setTimeout(() => toast.remove(), 300);
 }, 3000);
}

let currentModalProduct = null;

function openModal(i) {
 const p = renderedProducts[i];
 if (!p) return;
 currentModalProduct = p;

 if (typeof fbq !== 'undefined') {
   fbq('track', 'ViewContent', { content_name: p.name, content_category: p.category });
 }

 const hasDis = p.discountedPrice > 0 && p.discountedPrice < p.price;
 const showPrice = hasDis ? p.discountedPrice : p.price;
 const savings = hasDis ? (p.price - p.discountedPrice) : 0;
 const images = [p.thumbnail, p.image1, p.image2].filter(x => x && x.trim() !== "");
 const mainSrc = images[0] || "https://via.placeholder.com/400x400?text=11Cart";

 document.getElementById('modalMainImg').src = mainSrc;
 document.getElementById('modalName').textContent = p.name;
 document.getElementById('modalCat').textContent = p.category;
 document.getElementById('modalPrice').textContent = "Rs " + formatPrice(showPrice);

 const oldEl = document.getElementById('modalOldPrice');
 oldEl.style.display = hasDis ? "inline" : "none";
 if (hasDis) oldEl.textContent = "Rs " + formatPrice(p.price);

 const savEl = document.getElementById('modalSavings');
 savEl.style.display = hasDis ? "inline-flex" : "none";
 if (hasDis) savEl.textContent = `🔥 You save Rs ${formatPrice(savings)}!`;

 const tags = [];
 if (p.ageRange) tags.push("👶 " + p.ageRange);
 if (p.material) tags.push("🧱 " + p.material);
 if (p.theme) tags.push("🎯 " + p.theme);
 if (p.brand && p.brand.toLowerCase() !== "11 cart") tags.push("🏷️ " + p.brand);
 if (p.stock > 0 && p.stock <= 5) tags.push("⚡ Only " + p.stock + " left!");
 document.getElementById('modalTags').innerHTML = tags.map(t => `<span class="modal-tag">${t}</span>`).join("");

 document.getElementById('modalDesc').innerHTML = p.description ? p.description.replace(/\n/g, "<br>") : "<em>Contact us on WhatsApp for more details about this product.</em>";

 const thumbsEl = document.getElementById('modalThumbs');
 thumbsEl.innerHTML = images.length > 1

 document.getElementById('modalActions').innerHTML = `

 `;

 document.getElementById('modalOverlay').classList.add('open');
 document.body.style.overflow = 'hidden';
}
